Exemplos - Imagens

Efeito Escala de Cinza

from mediacomp.image import  * 

picture=makePicture(pickAFile())

for px in getPixels(picture): 
	novoVermelho = getRed(px) * 0.299
	novoVerde = getGreen(px) * 0.587
	novoAzul = getBlue(px) * 0.114
	luminancia = novoVermelho+novoVerde+novoAzul
	setColor(px,makeColor(luminancia,luminancia,luminancia))
	
show(picture)

Efeito Espelhado

from mediacomp.image import  *

imagem = picture=makePicture(pickAFile())
pontoespelhamento = int(getWidth(imagem)/2)

for y in range(1,getHeight(imagem)):
    for x in range(1,pontoespelhamento):
        p = getPixel(imagem, x+pontoespelhamento,y)
        p2 = getPixel(imagem, pontoespelhamento-x,y)
        setColor(p,makeColor(getRed(p2), getGreen(p2), getBlue(p2)))

show(imagem)

Textos, Linhas e Retângulos

from mediacomp.image import  *

imagem=makePicture(pickAFile())
cor = makeColor(255,255,255)

addRectFilled(imagem,  cor, 10, 300, 30, 310)
addText(imagem, cor, 70, 300, "Seymour Papert")
addRectFilled(imagem, cor , 200, 300, 220, 310)
addLine(imagem, cor, 30, 300, 200, 300)
addLine(imagem, cor, 30, 310, 200, 310)

show(imagem)

Efeito Preto e Branco

from mediacomp.image import  *

imagem=makePicture(pickAFile())

for x in range(1, getWidth(imagem)):
    for y in range(1, getHeight(imagem)):
        p = getPixel(imagem, x, y)
        if ((getRed(p) + getBlue(p) + getGreen(p)) / 255 < 1):
            setColor(p, makeColor(0, 0, 0))
        else:
            setColor(p, makeColor(255, 255, 255))

show(imagem)

Efeito Negativos

from mediacomp.image import  *

imagem=makePicture(pickAFile())

for px in getPixels(imagem):
    red = getRed(px)
    green = getGreen(px)
    blue = getBlue(px)
    negColor = makeColor(255 - red, 255 - green, 255 - blue)
    setColor(px, negColor)

show(imagem)

Efeito Chromakey

from mediacomp.image import  *

imagem=makePicture(pickAFile())
fundo=makePicture(pickAFile())

for x in range(1, getWidth(imagem)):
    for y in range(1, getHeight(imagem)):
        p = getPixel(imagem, x, y)
        if (getRed(p) + getBlue(p) < getGreen(p)):
            setColor(p, getColor(getPixel(fundo, x, y)))

show(imagem)
